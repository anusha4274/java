public class TriangleAreaCalculator {
                public static void main(String[] args) {
                    // Example base and height
                    double base = 10.0; // You can change this value to test with different bases
                    double height = 5.0; // You can change this value to test with different heights
            
                    // Call the method to calculate the area
                    double area = calculateArea(base, height);
            
                    // Output the result
                    System.out.println("The area of the triangle with base " + base + " and height " + height + " is: " + area);
                }
            
                // Method to calculate the area of a triangle
                public static double calculateArea(double base, double height) {
                    // Area formula: (base * height) / 2
                    return (base * height) / 2;
                }
            }
