abstract class Vehicle {
        public abstract void startEngine();
        public abstract void stopEngine();
    }
    
    class Car extends Vehicle {
        @Override
        public void startEngine() {
            System.out.println("The car's engine starts with a key turn.");
        }
    
        @Override
        public void stopEngine() {
            System.out.println("The car's engine stops when the key is turned off.");
        }
    }
    
    class Motorcycle extends Vehicle {
        @Override
        public void startEngine() {
            System.out.println("The motorcycle's engine starts with a push of a button.");
        }
    
        @Override
        public void stopEngine() {
            System.out.println("The motorcycle's engine stops when the button is pressed again.");
        }
    }
    
    public class Main {
        public static void main(String[] args) {
            Vehicle car = new Car();
            Vehicle motorcycle = new Motorcycle();
    
            System.out.println("Car behavior:");
            car.startEngine();
            car.stopEngine();
    
            System.out.println("\nMotorcycle behavior:");
            motorcycle.startEngine();
            motorcycle.stopEngine();
        }
    }

                    OUTPUT

                    Car behavior:
                    The car's engine starts with a key turn.
                    The car's engine stops when the key is turned off.
                    
                    Motorcycle behavior:
                    The motorcycle's engine starts with a push of a button.
                    The motorcycle's engine stops when the button is pressed again.
