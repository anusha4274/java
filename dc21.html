public class StringToBooleanExample {
                public static void main(String[] args) {
                    String trueString = "true";
                    String falseString = "false";
                    String otherString = "hello"; // Any other string
            
                    // Converting String to boolean
                    boolean bool1 = Boolean.parseBoolean(trueString);
                    boolean bool2 = Boolean.parseBoolean(falseString);
                    boolean bool3 = Boolean.parseBoolean(otherString);
            
                    System.out.println("Converted boolean from '" + trueString + "': " + bool1);   // true
                    System.out.println("Converted boolean from '" + falseString + "': " + bool2);  // false
                    System.out.println("Converted boolean from '" + otherString + "': " + bool3);   // false
                }
            }

Output


Converted boolean from 'true': true
Converted boolean from 'false': false
Converted boolean from 'hello': false
