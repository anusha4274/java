import java.util.Date;

    class MyDate extends Date {
        // Constructor to initialize MyDate object with the current date
        public MyDate() {
            super();  // Call the superclass constructor to initialize the date
        }
    
        // Method to return the current date as a String
        public String getCurrentDate() {
            return super.toString();  // Use the superclass method to convert Date to a String
        }
    }
    
    public class InheritedDateExample {
        public static void main(String[] args) {
            // Create an object of MyDate class, which inherits from Date
            MyDate myDate = new MyDate();
    
            // Print the current date
            System.out.println("Current Date: " + myDate.getCurrentDate());
        }
    }

    
Output


    Current Date: Thu Nov 07 11:13:25 IST 2024
