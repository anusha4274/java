public class CharToIntExample {
                public static void main(String[] args) {
                    char ch = 'A'; // Example character
            
                    // Method 1: ASCII/Unicode Value Conversion
                    int asciiValue = (int) ch;
            
                    // Method 2: Numeric Character Conversion (for '0' to '9' only)
                    char digitChar = '5';
                    int numericValue = digitChar - '0';
            
                    System.out.println("ASCII/Unicode value of '" + ch + "': " + asciiValue);
                    System.out.println("Numeric value of '" + digitChar + "': " + numericValue);
                }
            }

Output


ASCII/Unicode value of 'A': 65
Numeric value of '5': 5
