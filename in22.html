abstract class Person {
        public abstract void eat();
        public abstract void exercise();
    }
    
    class Athlete extends Person {
        @Override
        public void eat() {
            System.out.println("The athlete eats a high-protein meal to fuel the body.");
        }
    
        @Override
        public void exercise() {
            System.out.println("The athlete trains intensely with daily workouts and physical conditioning.");
        }
    }
    
    class LazyPerson extends Person {
        @Override
        public void eat() {
            System.out.println("The lazy person eats fast food while lounging on the couch.");
        }
    
        @Override
        public void exercise() {
            System.out.println("The lazy person avoids exercise and prefers to rest most of the day.");
        }
    }
    
    public class Main {
        public static void main(String[] args) {
            Person athlete = new Athlete();
            Person lazyPerson = new LazyPerson();
    
            System.out.println("Athlete's routine:");
            athlete.eat();
            athlete.exercise();
    
            System.out.println("\nLazy Person's routine:");
            lazyPerson.eat();
            lazyPerson.exercise();
        }
    }

                 OUTPUT

                 Athlete's routine:
                 The athlete eats a high-protein meal to fuel the body.
                 The athlete trains intensely with daily workouts and physical conditioning.
                 
                 Lazy Person's routine:
                 The lazy person eats fast food while lounging on the couch.
                 The lazy person avoids exercise and prefers to rest most of the day.
