interface Encryptable {
        String encrypt(String data);
        String decrypt(String encryptedData);
    }
    
    class AES implements Encryptable {
        @Override
        public String encrypt(String data) {
            // Simple simulation of encryption (in real-world AES uses complex algorithms)
            StringBuilder encrypted = new StringBuilder();
            for (char c : data.toCharArray()) {
                encrypted.append((char)(c + 1)); // Simple encryption: shift each character by 1
            }
            return encrypted.toString();
        }
    
        @Override
        public String decrypt(String encryptedData) {
            // Simple simulation of decryption
            StringBuilder decrypted = new StringBuilder();
            for (char c : encryptedData.toCharArray()) {
                decrypted.append((char)(c - 1)); // Simple decryption: shift each character back by 1
            }
            return decrypted.toString();
        }
    }
    
    class RSA implements Encryptable {
        @Override
        public String encrypt(String data) {
            // Simple simulation of encryption (in real-world RSA uses public key cryptography)
            StringBuilder encrypted = new StringBuilder();
            for (char c : data.toCharArray()) {
                encrypted.append((char)(c + 2)); // Simple encryption: shift each character by 2
            }
            return encrypted.toString();
        }
    
        @Override
        public String decrypt(String encryptedData) {
            // Simple simulation of decryption
            StringBuilder decrypted = new StringBuilder();
            for (char c : encryptedData.toCharArray()) {
                decrypted.append((char)(c - 2)); // Simple decryption: shift each character back by 2
            }
            return decrypted.toString();
        }
    }
    
    public class Main {
        public static void main(String[] args) {
            Encryptable aes = new AES();
            Encryptable rsa = new RSA();
    
            String originalData = "Hello World!";
            
            // AES encryption and decryption
            String aesEncrypted = aes.encrypt(originalData);
            System.out.println("AES Encrypted: " + aesEncrypted);
            String aesDecrypted = aes.decrypt(aesEncrypted);
            System.out.println("AES Decrypted: " + aesDecrypted);
    
            // RSA encryption and decryption
            String rsaEncrypted = rsa.encrypt(originalData);
            System.out.println("RSA Encrypted: " + rsaEncrypted);
            String rsaDecrypted = rsa.decrypt(rsaEncrypted);
            System.out.println("RSA Decrypted: " + rsaDecrypted);
        }
    }
