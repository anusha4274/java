public class LongestRepeatingSequence {
    
            // Function to find the longest repeating sequence in the string
            public static String longestRepeatingSequence(String str) {
                int n = str.length();
                String longestSequence = "";
                
                // Use a nested loop to generate all possible substrings
                for (int i = 0; i < n; i++) {
                    for (int j = i + 1; j <= n; j++) {
                        String subString = str.substring(i, j);
                        
                        // If the substring repeats and is longer than the previous longest
                        if (str.indexOf(subString, i + 1) != -1 && subString.length() > longestSequence.length()) {
                            longestSequence = subString;
                        }
                    }
                }
                
                return longestSequence;
            }
        
            public static void main(String[] args) {
                String input = "ABABAB";
                String result = longestRepeatingSequence(input);
                
                System.out.println("The longest repeating sequence is: " + result);
            }
        }
