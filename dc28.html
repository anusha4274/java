public class DecimalToHexExample {
                public static void main(String[] args) {
                    int decimalNumber = 419; // Example decimal number
            
                    // Using Integer.toHexString to convert decimal to hexadecimal
                    String hexString = Integer.toHexString(decimalNumber).toUpperCase();
            
                    System.out.println("Decimal: " + decimalNumber);
                    System.out.println("Hexadecimal: " + hexString);
                }
            }

Output


Decimal: 419
Hexadecimal: 1A3
