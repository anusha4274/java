import java.util.Arrays;

    class SortThread extends Thread {
        private int[] array;
        private int start, end;
    
        public SortThread(int[] array, int start, int end) {
            this.array = array;
            this.start = start;
            this.end = end;
        }
    
        public void run() {
            Arrays.sort(array, start, end);
        }
    }
    
    public class MultiThreadedSort {
        public static void main(String[] args) throws InterruptedException {
            int[] arr = {12, 11, 13, 5, 6, 7, 9, 2, 1, 3, 8, 4};
    
            int numberOfThreads = 4;
            int chunkSize = arr.length / numberOfThreads;
            SortThread[] threads = new SortThread[numberOfThreads];
    
            for (int i = 0; i < numberOfThreads; i++) {
                int start = i * chunkSize;
                int end = (i == numberOfThreads - 1) ? arr.length : (i + 1) * chunkSize;
                threads[i] = new SortThread(arr, start, end);
                threads[i].start();
            }
    
            for (int i = 0; i < numberOfThreads; i++) {
                threads[i].join();
            }
    
            Arrays.sort(arr);  // Merge the sorted subarrays
    
            System.out.println("Sorted array: " + Arrays.toString(arr));
        }
    }
