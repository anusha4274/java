import java.util.Arrays;

            public class SwapFirstLast {
                public static void main(String[] args) {
                    // Original array
                    int[] originalArray = {1, 2, 3, 4, 5};
            
                    // Check if the array has at least two elements
                    if (originalArray.length < 2) {
                        System.out.println("Array must have at least two elements to swap.");
                        return;
                    }
            
                    // Create a new array for the result
                    int[] newArray = Arrays.copyOf(originalArray, originalArray.length);
            
                    // Swap the first and last elements
                    newArray[0] = originalArray[originalArray.length - 1]; // Set the first element to the last
                    newArray[newArray.length - 1] = originalArray[0]; // Set the last element to the first
            
                    // Display the original and new arrays
                    System.out.println("Original Array: " + Arrays.toString(originalArray));
                    System.out.println("New Array after swapping first and last elements: " + Arrays.toString(newArray));
                }
            }

Output


Original Array: [1, 2, 3, 4, 5]
New Array after swapping first and last elements: [5, 2, 3, 4, 1]
