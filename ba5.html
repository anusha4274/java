import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.IOException;

public class AverageOfFiveNumbers {
    public static void main(String[] args) throws IOException {
        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));

        System.out.print("Enter the first number: ");
        double num1 = Double.parseDouble(reader.readLine());

        System.out.print("Enter the second number: ");
        double num2 = Double.parseDouble(reader.readLine());

        System.out.print("Enter the third number: ");
        double num3 = Double.parseDouble(reader.readLine());

        System.out.print("Enter the fourth number: ");
        double num4 = Double.parseDouble(reader.readLine());

        System.out.print("Enter the fifth number: ");
        double num5 = Double.parseDouble(reader.readLine());

        double sum = num1 + num2 + num3 + num4 + num5;
        double average = sum / 5;

        System.out.println("The average of the five numbers is: " + average);
    }
}
