public class SumOfDigits {
                public static void main(String[] args) {
                    // Example positive integer
                    int number = 12345; // You can change this value to test with different integers
            
                    // Check if the number is positive
                    if (number < 0) {
                        System.out.println("Error: Please provide a positive integer.");
                        return;
                    }
            
                    // Call the method to calculate the sum of digits
                    int sum = sumOfDigits(number);
            
                    // Output the result
                    System.out.println("The sum of the digits of " + number + " is: " + sum);
                }
            
                // Method to calculate the sum of digits
                public static int sumOfDigits(int num) {
                    int sum = 0;
            
                    // Loop to extract each digit and add to sum
                    while (num > 0) {
                        sum += num % 10; // Add the last digit to sum
                        num /= 10;       // Remove the last digit
                    }
            
                    return sum;
                }
            }

Output


The sum of the digits of 12345 is: 15
