class Str47 {
            public static void main(String[] args) {
                System.out.print("Enter the first string: ");
                String str1 = System.console().readLine();
        
                System.out.print("Enter the second string: ");
                String str2 = System.console().readLine();
        
                if (areSameCharacters(str1, str2)) {
                    System.out.println("The two strings are created with the same characters.");
                } else {
                    System.out.println("The two strings are not created with the same characters.");
                }
            }
        
            public static boolean areSameCharacters(String str1, String str2) {
                if (str1.length() != str2.length()) {
                    return false;
                }
        
                int[] count1 = new int[256]; // assuming ASCII characters
                int[] count2 = new int[256];
        
                for (int i = 0; i < str1.length(); i++) {
                    count1[str1.charAt(i)]++;
                    count2[str2.charAt(i)]++;
                }
        
                for (int i = 0; i < 256; i++) {
                    if (count1[i] != count2[i]) {
                        return false;
                    }
                }
        
                return true;
            }
        }
        
        OUTPUT

        Enter the second string: mad
        The two strings are created with the same characters.
