import java.sql.Timestamp;
import java.util.Date;

public class TimestampToDateExample {
    public static void main(String[] args) {
        // Creating a Timestamp object
        Timestamp timestamp = new Timestamp(System.currentTimeMillis()); // Current timestamp

        // Method 1: Using the Date constructor
        Date dateFromTimestamp = new Date(timestamp.getTime());

        // Method 2: Directly using Timestamp as Date
        Date dateDirect = new Date(timestamp); // Implicit conversion

        System.out.println("Timestamp: " + timestamp);
        System.out.println("Converted Date (using constructor): " + dateFromTimestamp);
        System.out.println("Converted Date (using direct conversion): " + dateDirect);
    }
}

Output


Timestamp: 2024-11-04 12:34:56.123456789
Converted Date (using constructor): Mon Nov 04 12:34:56 GMT 2024
Converted Date (using direct conversion): Mon Nov 04 12:34:56 GMT 2024
