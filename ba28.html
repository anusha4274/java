public class RectangleAreaCalculator {
                public static void main(String[] args) {
                    // Example length and width
                    double length = 10.0; // You can change this value to test with different lengths
                    double width = 5.0;   // You can change this value to test with different widths
            
                    // Call the method to calculate the area
                    double area = calculateArea(length, width);
            
                    // Output the result
                    System.out.println("The area of the rectangle with length " + length + " and width " + width + " is: " + area);
                }
            
                // Method to calculate the area of a rectangle
                public static double calculateArea(double length, double width) {
                    // Area formula: length * width
                    return length * width;
                }
            }
