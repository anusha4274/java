public class BooleanToStringExample {
                public static void main(String[] args) {
                    boolean boolValue1 = true;
                    boolean boolValue2 = false;
            
                    // Method 1: Using String.valueOf()
                    String strValue1 = String.valueOf(boolValue1);
                    String strValue2 = String.valueOf(boolValue2);
            
                    // Method 2: Using Boolean.toString()
                    String strValue3 = Boolean.toString(boolValue1);
                    String strValue4 = Boolean.toString(boolValue2);
            
                    System.out.println("String from boolean true: " + strValue1);  // "true"
                    System.out.println("String from boolean false: " + strValue2); // "false"
                    System.out.println("String from boolean true (Boolean.toString): " + strValue3);  // "true"
                    System.out.println("String from boolean false (Boolean.toString): " + strValue4); // "false"
                }
            }

Output


String from boolean true: true
String from boolean false: false
String from boolean true (Boolean.toString): true
