public class BinaryToDecimalExample {
                public static void main(String[] args) {
                    String binaryString = "1011"; // Example binary number
                    int decimalValue = 0;
            
                    // Manual conversion from binary to decimal
                    for (int i = 0; i < binaryString.length(); i++) {
                        char bit = binaryString.charAt(binaryString.length() - 1 - i); // Get the bit from right to left
                        if (bit == '1') {
                            decimalValue += Math.pow(2, i); // Add 2 raised to the power of the bit position
                        }
                    }
            
                    System.out.println("Binary: " + binaryString);
                    System.out.println("Decimal: " + decimalValue);
                }
            }

Output


Binary: 1011
Decimal: 11
