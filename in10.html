// Base class Employee
        class Employee {
            protected String name;
            protected String address;
            protected double salary;
            protected String jobTitle;
        
            public Employee(String name, String address, double salary, String jobTitle) {
                this.name = name;
                this.address = address;
                this.salary = salary;
                this.jobTitle = jobTitle;
            }
        
            public double calculateBonus() {
                return salary * 0.05;
            }
        
            public String generatePerformanceReport() {
                return "Performance report for " + name + " (" + jobTitle + ")";
            }
        
            public String manageProject(String projectName) {
                return name + " is managing the project: " + projectName;
            }
        
            public void printEmployeeInfo() {
                System.out.println("Name: " + name);
                System.out.println("Address: " + address);
                System.out.println("Salary: $" + salary);
                System.out.println("Job Title: " + jobTitle);
            }
        }
        
        // Manager subclass
        class Manager extends Employee {
        
            public Manager(String name, String address, double salary) {
                super(name, address, salary, "Manager");
            }
        
            @Override
            public double calculateBonus() {
                return salary * 0.15;
            }
        
            @Override
            public String manageProject(String projectName) {
                return name + " (Manager) is overseeing the project: " + projectName;
            }
        }
        
        // Developer subclass
        class Developer extends Employee {
        
            public Developer(String name, String address, double salary) {
                super(name, address, salary, "Developer");
            }
        
            @Override
            public double calculateBonus() {
                return salary * 0.10;
            }
        
            @Override
            public String manageProject(String projectName) {
                return name + " (Developer) is developing the project: " + projectName;
            }
        }
        
        // Programmer subclass
        class Programmer extends Employee {
        
            public Programmer(String name, String address, double salary) {
                super(name, address, salary, "Programmer");
            }
        
            @Override
            public double calculateBonus() {
                return salary * 0.08;
            }
        
            @Override
            public String manageProject(String projectName) {
                return name + " (Programmer) is coding for the project: " + projectName;
            }
        }
        
        // Main class to test the hierarchy
        public class Main {
            public static void main(String[] args) {
                Manager manager = new Manager("Alice", "123 Maple St", 85000);
                Developer developer = new Developer("Bob", "456 Oak St", 75000);
                Programmer programmer = new Programmer("Charlie", "789 Pine St", 65000);
        
                System.out.println("Manager Info:");
                manager.printEmployeeInfo();
                System.out.println("Bonus: $" + manager.calculateBonus());
                System.out.println(manager.generatePerformanceReport());
                System.out.println(manager.manageProject("Project Alpha") + "\n");
        
                System.out.println("Developer Info:");
                developer.printEmployeeInfo();
                System.out.println("Bonus: $" + developer.calculateBonus());
                System.out.println(developer.generatePerformanceReport());
                System.out.println(developer.manageProject("Project Beta") + "\n");
        
                System.out.println("Programmer Info:");
                programmer.printEmployeeInfo();
                System.out.println("Bonus: $" + programmer.calculateBonus());
                System.out.println(programmer.generatePerformanceReport());
                System.out.println(programmer.manageProject("Project Gamma"));
            }
        }
