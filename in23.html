abstract class Instrument {
        public abstract void play();
        public abstract void tune();
    }
    
    class Glockenspiel extends Instrument {
        @Override
        public void play() {
            System.out.println("The Glockenspiel plays a series of clear, bell-like tones.");
        }
    
        @Override
        public void tune() {
            System.out.println("The Glockenspiel is tuned by adjusting the metal bars to the correct pitch.");
        }
    }
    
    class Violin extends Instrument {
        @Override
        public void play() {
            System.out.println("The Violin produces a smooth, melodic sound by drawing a bow across its strings.");
        }
    
        @Override
        public void tune() {
            System.out.println("The Violin is tuned by adjusting the tension of its strings using the tuning pegs.");
        }
    }
    
    public class Main {
        public static void main(String[] args) {
            Instrument glockenspiel = new Glockenspiel();
            Instrument violin = new Violin();
    
            System.out.println("Glockenspiel:");
            glockenspiel.play();
            glockenspiel.tune();
    
            System.out.println("\nViolin:");
            violin.play();
            violin.tune();
        }
    }

               OUTPUT

               Glockenspiel:
               The Glockenspiel plays a series of clear, bell-like tones.
               The Glockenspiel is tuned by adjusting the metal bars to the correct pitch.
               
               Violin:
               The Violin produces a smooth, melodic sound by drawing a bow across its strings.
               The Violin is tuned by adjusting the tension of its strings using the tuning pegs.
