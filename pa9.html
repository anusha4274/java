package c4.sample;

    public class SampleClass {
        // Public member variable
        public String publicField = "This is a public field";
    
        // Protected member variable
        protected String protectedField = "This is a protected field";
    
        // Public method
        public void publicMethod() {
            System.out.println("This is a public method");
        }
    
        // Protected method
        protected void protectedMethod() {
            System.out.println("This is a protected method");
        }
    }

    package c4.main;

import c4.sample.SampleClass;

public class MainClass {
    public static void main(String[] args) {
        // Create an object of SampleClass
        SampleClass sample = new SampleClass();
        
        // Access the public member directly
        System.out.println(sample.publicField);
        sample.publicMethod();

        // Access the protected member and method
        // Protected members can be accessed in the same package or through inheritance
        // Since MainClass is in a different package, we need inheritance or an instance of subclass
        SubClass subClass = new SubClass();
        subClass.accessProtectedMembers();
    }
}

// Subclass to access protected members from SampleClass
class SubClass extends SampleClass {
    public void accessProtectedMembers() {
        // Access protected field and method from SampleClass
        System.out.println(protectedField);
        protectedMethod();
    }
}
