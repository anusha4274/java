abstract class Bird {
        public abstract void fly();
        public abstract void makeSound();
    }
    
    class Eagle extends Bird {
        @Override
        public void fly() {
            System.out.println("The eagle soars high in the sky with strong, wide wings.");
        }
    
        @Override
        public void makeSound() {
            System.out.println("The eagle makes a sharp, high-pitched screech.");
        }
    }
    
    class Hawk extends Bird {
        @Override
        public void fly() {
            System.out.println("The hawk flies in a circular pattern, often gliding in the air.");
        }
    
        @Override
        public void makeSound() {
            System.out.println("The hawk makes a sharp, shrill scream.");
        }
    }
    
    public class Main {
        public static void main(String[] args) {
            Bird eagle = new Eagle();
            Bird hawk = new Hawk();
    
            System.out.println("Eagle:");
            eagle.fly();
            eagle.makeSound();
    
            System.out.println("\nHawk:");
            hawk.fly();
            hawk.makeSound();
        }
    }
 
                  OUTPUT

                  Eagle:
                  The eagle soars high in the sky with strong, wide wings.
                  The eagle makes a sharp, high-pitched screech.
                  
                  Hawk:
                  The hawk flies in a circular pattern, often gliding in the air.
                  The hawk makes a sharp, shrill scream.
