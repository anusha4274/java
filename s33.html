class Str33 {
                    public static void iterateOverCharacters(String str) {
                        for (int i = 0; i < str.length(); i++) {
                            char c = str.charAt(i);
                            System.out.println("Character at index " + i + ": " + c);
                        }}
                    public static void main(String[] args) {
                        String str = "Hello World!";
                        System.out.println("Iterating over characters using a for loop:");
                        iterateOverCharacters(str);
                    }}
